using AutoMapper;
using Domain.Entity;
using Infrastructure.Persistence.Repositories;
using Infrastructure.Wrappers;
using Microsoft.AspNetCore.Identity;
using Moq;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace xUnitTesting
{
    public class UserManagerWrapperUT
    {

        [Fact]
        public async Task AddToRolesAsync_ShouldSucceed_WhenUserExists()
        {
            //arrange
            var mockUserManager = new Mock<UserManager<ApplicationUser>>();
            var mockMapper = new Mock<IMapper>();
            var userManagerWrapper = new UserManagerWrapper(mockUserManager.Object, mockMapper.Object);
            var user = new User { Id = Guid.NewGuid().ToString() };
            var roles = new List<string> { Role.Admin.ToString() };

            mockUserManager.Setup(x => x.FindByIdAsync(user.Id)).ReturnsAsync(new ApplicationUser { Id = user.Id });
            mockUserManager.Setup(x => x.AddToRolesAsync(It.IsAny<ApplicationUser>(), roles)).ReturnsAsync(IdentityResult.Success);

            //Act
            var result = await userManagerWrapper.AddToRolesAsync(user, roles);

            //Assert
            Assert.NotNull(result);
            Assert.True(result.Succeeded);
        }
    }
}
